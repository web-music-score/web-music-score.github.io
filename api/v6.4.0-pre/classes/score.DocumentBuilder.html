    Class DocumentBuilder

    Document builder class.

    // Example
    let doc = new Score.DocumentBuilder()
        .addScoreConfiguration({ type: "staff", clef: "G", isOctavewDown: true })
        .setMeasuresPerRow(4)
        .addMeasure()
        .addNote(1, "C3", "4n")
        .addChord(1, ["C3", "E3", "G3"], "4n").addAnnotation("chordLabel", "C")
        .addRest(1, "4n")
        // etc.
        .getDEocument();
